CREATE TABLE PageDatas (
	id INTEGER AUTO_INCREMENT,
	title CHAR(128) NOT NULL,
	content CHAR(1024) NOT NULL,
	PRIMARY KEY (id)
)

CREATE TABLE Menus (
	menuId INTEGER AUTO_INCREMENT,
	id INTEGER NOT NULL,
	priorityNumber INTEGER NOT NULL,
	title CHAR(128) NOT NULL,
	description CHAR(512) NOT NULL,
	tileImgURL CHAR(128) NOT NULL,
	sliderImgURL CHAR(128) NOT NULL,
	PRIMARY KEY (menuId, id)
	FOREIGN KEY (id) REFERENCES PageDatas (id)
				ON DELETE CASCADE
				ON UPDATE NO ACTION
)

CREATE TABLE Booklets (
	bookletId INTEGER AUTO_INCREMENT,
	id INTEGER  NOT NULL,
	PRIMARY KEY (bookletId, id)
	FOREIGN KEY (id) REFERENCES PageDatas (id)
				ON DELETE CASCADE
				ON UPDATE NO ACTION
)


CREATE TABLE Chapters (
	chapterId INTEGER AUTO_INCREMENT,
	id INTEGER NOT NULL,
	PRIMARY KEY (chapterId, id)
	FOREIGN KEY (id) REFERENCES PageDatas (id)
)


CREATE TABLE Pages (
	pageId INTEGER AUTO_INCREMENT,
	id INTEGER NOT NULL,
	PRIMARY KEY (pageId, id)
	FOREIGN KEY (id) REFERENCES PageDatas (id)
				ON DELETE CASCADE
				ON UPDATE NO ACTION
)


CREATE TABLE Resources (
	id INTEGER AUTO_INCREMENT,
	title CHAR(128) NOT NULL,
	description CHAR(512) NOT NULL,
	resourceURL CHAR(128) NOT NULL,
	PRIMARY KEY (id)
)
